	\subsubsection*{Dictionaries}
	
	An example:
	\begin{lstlisting}
ages = {"Alice": 42, "Bob": 40}\end{lstlisting}
	Indexing:
	\begin{lstlisting}
>>> ages["Bob"]
40\end{lstlisting}
	Keys:
	\begin{lstlisting}
>>> ages.keys()
dict_keys(['Alice', 'Bob'])\end{lstlisting}
	Adding or changing items:
	\begin{lstlisting}
>>> ages.update({"Charlie": 1})\end{lstlisting}
